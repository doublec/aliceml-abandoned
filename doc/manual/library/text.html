An extended version of the Standard ML Basis' TEXT signature.

See also: CHAR, STRING, SUBSTRING, MONO_VECTOR, MONO_ARRAY, MONO_VECTOR_SLICE, MONO_ARRAY_SLICE

Imported implicitly. 

    signature TEXT =
    sig
	structure Char :            CHAR
	structure String :          STRING
				    where type char   = Char.t
				      and type string = Char.string
	structure Substring :       SUBSTRING
				    where type char   = Char.t
				      and type string = String.t
	structure CharVector :      MONO_VECTOR
				    where type elem   = Char.t
				      and type vector = String.t
	structure CharArray :       MONO_ARRAY
				    where type elem   = Char.t
				      and type vector = String.t
	structure CharVectorSlice : MONO_VECTOR_SLICE
				    where type elem   = Char.t
				      and type vector = String.t
	structure CharArraySlice :  MONO_ARRAY_SLICE
				    where type elem   = Char.t
				      and type vector = String.t
				      and type array  = CharArray.t
				      and type vector_slice = CharVectorSlice.t
    end
